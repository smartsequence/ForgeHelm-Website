  private cache: TranslationCache = {};
  private apiUrl = import.meta.env.PUBLIC_API_URL;

  async getContent(key: string, lang: string): Promise<string> {
    // æª¢æŸ¥æœ¬åœ°å¿«å–

    // å‘¼å« API

        `${this.apiUrl}/api/content/${key}?lang=${lang}`
      );
      const data = await response.json();
      
      // å„²å­˜åˆ°æœ¬åœ°å¿«å–
      if (!this.cache[key]) {
        this.cache[key] = {};

      this.cache[key][lang] = data.content;
      
      return data.content;
    } catch (error) {
      console.error('Translation error:', error);
      // å‚™æ´ï¼šè¿”å›ç¹ä¸­
      return await this.getContent(key, 'zh-TW');
    }
  }

  async translateText(text: string, targetLang: string): Promise<string> {
    const response = await fetch(`${this.apiUrl}/api/translate`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        text,
        targetLang,
        sourceLang: 'zh-TW',
      }),
    });
    
    const data = await response.json();
    return data.translatedText;
  }
}

export const translation = new TranslationService();
